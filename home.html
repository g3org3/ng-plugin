    Angular Plugin

    navbar

    <navbar type='static' theme='red' title='webpage' href='#'></navbar>
// Params
// title    = text 
// type     = [static, fixed] 
// ng-theme = $scope value for theme
// href     = link for title
// theme    = [
	blue, red, green, purple, gray, white, inverse,
	blue-theme, red-theme, green-theme, purple-theme, gray-theme, white-theme, inverse-theme,
    ]

    menu

    <navbar type='static' theme='red' title='webpage' href='#'>
    <menu>
    	<li><a href="#link">option title</a></li>
    </menu>
</navbar>

    dropdown

    <navbar type='static' theme='red' title='webpage' href='#'>
    <menu>
    	<li><a href="#link">option title</a></li>
    	<dropdown title='' href=''>
    		<li><a href="#link">option title</a></li>
    	</dropdown>
    </menu>
</navbar>
